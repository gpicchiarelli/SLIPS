import SLIPS

// Create environment
CLIPS.createEnvironment()

// Define template
CLIPS.eval(expr: """
(deftemplate person
    (slot name)
    (slot age (type INTEGER)))
""")

// Define rule
CLIPS.eval(expr: """
(defrule greet-adult
    (person (name ?n) (age ?a&:(>= ?a 18)))
    =>
    (printout t "Hello " ?n crlf))
""")

// Run the engine
CLIPS.run()
